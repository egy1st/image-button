Tutorial

This tutorial describe most of features supported by DC.ImageButton

also you can refer to the project  example which installed by default into C:\Program Files\Dynamic Components\Image Button\Tutorial\

 

Dim IB As New DynamicComponents.ImageButton()

IB.InitForm(Me)

IB.PrepareImageButtons(Application.StartupPath + "\icons\", False)

IB.AddImageButton(Me.FirstButton, "first.ico", "firstover.ico", "firstdown.ico")

IB.AddImageButton(Me.PreviousButton, "previous.ico", "previousover.ico", "previousdown.ico")

IB.AddImageButton(Me.NextButtn, "next.ico", "nextover.ico", "nextdown.ico")

IB.AddImageButton(Me.LastButton, "last.ico", "lastover.ico", "lastdown.ico")
